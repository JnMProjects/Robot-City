# Avatar
<Primary />
<Collapse.Group>
    <Collapse.Object trigger="Code" title="">
        ## Avatar
        ### Usage
        ```tsx
        import { Avatar } from '@ui/.';

        <Avatar username="github" />
        ```

        ### Arguments
        username: string // The username of the user

        size: number // The size of the avatar in px

        placeholder: boolean // If the avatar should be a placeholder

        links: boolean // If the avatar should be a link to the user's profile

        ## AvatarGroup
        ### Usage
        ```tsx
        import { AvatarGroup } from '@ui/.';

        <AvatarGroup users={["derjacklive", "octocat", "github"]} />
        ```

        ### Arguments
        users: string[] // The usernames of the users

        size: number // The size of the avatar in px

        limit: number // The maximum amount of avatars to display // Rest will be displayed as a number

        <Popup trigger="Limit" >
            <AvatarGroup users={["jackatdjl", "octocat", "github", "mutebefehl"]} limit={3} />
        </Popup>

    </Collapse.Object>

</Collapse.Group>
